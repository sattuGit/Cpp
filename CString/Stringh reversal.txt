#include <iostream>
#include <stack>
#include <string.h>
#define MAXLEN 15
using namespace std;


void reverseSTL(char *list){
    char *t=list;
    stack<char> pool;
    while(*t!='\0'){
        pool.push(*t);
        ++t;
    }
    while(!pool.empty()){
        *list=pool.top();
        ++list;
        pool.pop();
    }
    cout<<endl;
}


void reversal(char *d, size_t l){
    char c;
    for(int i=0;i<(l/2);++i){
        c =*(d+i);
        *(d+i)=*(d+l-1-i);
        *(d+l-1-i)=c;
    }
}

int main()
{
    char name[]="satendrak";
    
    cout<<"_1"<<name<<"_"<<endl;
    reverseSTL(name);
    cout<<"_"<<name<<"_"<<endl;
    reversal(name,strlen(name));
    cout<<"_"<<name<<"_"<<endl;

    return 0;
}
